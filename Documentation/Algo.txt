DEBUT
	variable nombre_mystere : integer <- 90
	variable saisie : integer <- 0
	
	print("Saisissez une valeur entre 1 et 100: ")
	input(saisie)
	
	IF saisie < nombre_mystere
		print("Le nombre à trouver est supérieur.")
	FI
	IF saisie > nombre_mystere
		print("Le nombre à trouver est inférieur.")
	FI

		print("C'est gagné !")
	FI

	recommencer ce qui suit
		si le nombre saisit est inférieur au nombre mystère
		afficher le nombre à trouver est supérieur

		afficher le nombre à trouver est inférieur

	je sors de la boucle

DEBUT
	variable nombre_mystere : integer <- 90
	variable saisie : integer <- 0
	
	print("Saisissez une valeur entre 1 et 100: ")
	input(saisie)
	
	WHILE saisie != nombre_mystere
		IF saisie < nombre_mystere
			print("Le nombre à trouver est supérieur.")
		ELSE
			print("Le nombre à trouver est inférieur.")
		FI
		
		print("Saisissez une valeur entre 1 et 100: ")
		input(saisie)
	END WHILE
	IF saisie == nombre_mystere
		print("C'est gagné !")
	FI
FIN
